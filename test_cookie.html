Otwórz tę stronę na claude.ai

1. Zaloguj się do claude.ai

2. Otwórz DevTools (F12) → Console

3. Wklej ten kod:

// Sprawdź czy sessionKey jest dostępne
console.log("Wszystkie cookies:", document.cookie);
const sk = document.cookie.split(';').find(c => c.includes('sessionKey'));
if (sk) {
    const value = sk.split('=').slice(1).join('=').trim();
    console.log("sessionKey:", value);
    // Skopiuj do schowka
    navigator.clipboard.writeText(value).then(() => {
        alert("sessionKey skopiowane do schowka!");
    });
} else {
    console.log("sessionKey nie jest dostępne przez JavaScript (HttpOnly)");
    alert("sessionKey jest HttpOnly - musisz skopiować ręcznie z Application → Cookies");
}
